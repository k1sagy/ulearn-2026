        <<interface>>

        <<abstract>>
        +GetCommand() TCommand
    }

    class Device~TCommand~ {
        <<abstract>>
        +ExecuteCommand(command) string
    }

    class ShooterAI {
        <<class>>
        -int counter
        +GetCommand() IShooterMoveCommand
    }

    class BuilderAI {
        <<class>>

        +GetCommand() IMoveCommand
    }

    class Mover {
        <<class>>
        +ExecuteCommand(command) string

        <<class>>

        <<class>>
        -IRobotAI~TCommand~ ai

        <<interface>>
        +Point Destination
    }

    class IShooterMoveCommand {
        <<interface>>
        +bool ShouldHide
    }

    class ShooterCommand {
        <<class>>
        +Point Destination
        +bool Shoot
        +bool ShouldHide
    }

    class BuilderCommand {
        <<class>>
        +Point Destination
        +bool Build
    }

        +double Y

    IMoveCommand <|.. BuilderCommand : реализует

    BuilderCommand --> Point : Destination

    Robot~TCommand~ o-- IDevice~TCommand~ : содержит
