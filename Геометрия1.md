        <<class>>
        +double Radius
        +ContainsPoint(point) bool
        +GetBoundingBox() RectangularCuboid

        <<class>>
        +double SizeX

    }

    class Cylinder {

        +double SizeZ
        +double Radius

    }

    class CompoundBody {
        <<class>>
        +IReadOnlyList~Body~ Parts
        +ContainsPoint(point) bool

    }

        <<struct>>
        +double X
        +double Y
        +double Z
    }
